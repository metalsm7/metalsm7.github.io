<div id="app_event_scroll_ex1">
    <h3>Scroll (scroll, scroll.hit.top, scroll.hit.bottom)</h4>
    <div class="example">
        <div hc:scroll="onScroll" hc:scroll.hit.top="onTop" hc:scroll.hit.bottom="onBottom" style="width: 300px; height: 100px; overflow-y: scroll; border: 1px solid gray;">
            TOP<hr />
            1<br />2<br />3<br />4<br />5<hr />
            1<br />2<br />3<br />4<br />5<hr />
            1<br />2<br />3<br />4<br />5<hr />
            1<br />2<br />3<br />4<br />5<hr />
            BOTTOM
        </div>
        <br />
        <label hc:html="label"></label>
    </div>
</div>

<script type="text/javascript">
new Hacci({
    el: document.querySelector('#app_event_scroll_ex1'),
    data: {
        label: '',
        onTopOrBottom: false
    },
    method: {
        onScroll: function() {
            if (this.onTopOrBottom) return;
            this.label = '<div style="color: #666666;">on scrolling...</div>';
        },
        onTop: function() {
            // to display message for a while
            this.onTopOrBottom = true;
            setTimeout(function() { this.onTopOrBottom = false; }.bind(this), 200);
            //
            this.label = '<div style="color: #ff0000; font-weight: bold;">scroll top hit!</div>';
        },
        onBottom: function() {
            // to display message for a while
            this.onTopOrBottom = true;
            setTimeout(function() { this.onTopOrBottom = false; }.bind(this), 200);
            //
            this.label = '<div style="color: #0000ff; font-weight: bold;">scroll bottom hit!</div>';
        }
    }
}).mount();
</script>

<div id="app_for_ex1">
    <h3>For</h4>
    <div class="example">
        <table>
            <th>
                <tr>
                    <td colspan="4"> > list - 1</td>
                </tr>
            </th>
            <tr hc:for="data in list">
                <td hc:click="onOpen('id', data.id)" hc:text="data.id"></td>
                <td>{{ data.pwd }}</td>
                <td hc:click="onOpen('name', data.name)" hc:html="data.name"></td>
                <td hc:click="onRemove(data)">X</td>
            </tr>
        </table>
        <hr />
        <table>
            <th>
                <tr>
                    <td colspan="4"> > list - 2</td>
                </tr>
            </th>
            <tr hc:for="(id, pwd, name) in list2">
                <td hc:click="onOpen('id', id)" hc:text="id"></td>
                <td>{{ pwd }}</td>
                <td hc:click="onOpen('name', name)" hc:html="name"></td>
                <td hc:click="onRemoveById(id)">X</td>
            </tr>
        </table>
        <hr />
        <table>
            <tr>
                <td><input type="text" hc:model="req_data.id" style="width:50px;"></td>
                <td><input type="password" hc:model="req_data.pwd" style="width:50px;"></td>
                <td><input type="text" hc:model="req_data.name" style="width:50px;"></td>
                <td><button hc:click="onAdd(1)">+ list1</button></td>
                <td><button hc:click="onAdd(2)">+ list2</button></td>
            </tr>
        </table>
    </div>
</div>

<script type="text/javascript">
new Hacci({
    id: 'for',
    el: document.querySelector('#app_for_ex1'),
    data: {
        list: [
            { id: 'id1', pwd: 'pwd1', name: '<p style="color:red;">name1</p>', }, 
            { id: 'id2', pwd: 'pwd2', name: '<p style="color:green;">name2</p>', }, 
            { id: 'id3', pwd: 'pwd3', name: '<p style="color:tellow;">name3</p>', }, 
            { id: 'id4', pwd: 'pwd4', name: '<p style="color:blue;">name4</p>', }, 
        ],
        list2: [
            { id: 'id11', pwd: 'pwd11', name: '<p style="color:red;">name11</p>', }, 
            { id: 'id12', pwd: 'pwd12', name: '<p style="color:green;">name12</p>', }, 
        ],
        req_data: {
            id: '', 
            pwd: '',
            name: '',
        }
    },
    method: {
        onOpen: function(type, data) {
            alert(type + ' is ' + data);
        },
        onRemove: function(row) {
            this.list.indexOf(row) > -1 && this.list.splice(this.list.indexOf(row), 1);
        },
        onRemoveById: function(id) {
            for (let cnti = 0; cnti < this.list2.length; cnti++) {
                if (this.list2[cnti].id === id) {
                    this.list2.splice(cnti, 1);
                    cnti--;
                }
            }
        },
        onAdd: function(target) {
            if (target === 1) {
                this.list.push(JSON.parse(JSON.stringify(this.req_data)));
            }
            else {
                this.list2.push(JSON.parse(JSON.stringify(this.req_data)));
            }
            this.clearReqData();
        },
        clearReqData: function() {
            this.req_data.id = '';
            this.req_data.pwd = '';
            this.req_data.name = '';
        },
    }
}).mount();
</script>
