<div id="app_event_click_ex1">
    <h3>Click</h4>
    <div class="example">
        <button hc:click="onClick('#1', _event)">#1</button>
        <button hc:click="onClick('#2', _event)">#2</button>
        <div hc:click="onClick('#3', _event)" style="border: 1px solid red; background-color: #bcbcbc; width: 100px; height: 20px;">#3</div>
        <br />
        <label hc:ref="label"></label>
    </div>
</div>

<script>
new Hacci({
    el: document.querySelector('#app_event_input_ex1'),
    data: {
        val: null
    },
    method: {
        onClick: function(val, e) {
            this.refs.label.innerHTML = val + ' clicked - ' + e.target.outerHTML.replace(//, '>');
        }
    }
}).mount();
</script>

<div id="app_event_scroll_ex1">
    <h3>Scroll (scroll, scroll.hit.top, scroll.hit.bottom)</h4>
    <div class="example">
        <div hc:scroll="onScroll" hc:scroll.hit.top="onTop" hc:scroll.hit.bottom="onBottom" style="width: 300px; height: 100px; overflow-y: scroll; border: 1px solid gray;">
            TOP<hr />
            1<br />2<br />3<br />4<br />5<hr />
            1<br />2<br />3<br />4<br />5<hr />
            1<br />2<br />3<br />4<br />5<hr />
            1<br />2<br />3<br />4<br />5<hr />
            BOTTOM
        </div>
        <br />
        <label hc:html="label"></label>
    </div>
</div>

<script type="text/javascript">
new Hacci({
    el: document.querySelector('#app_event_scroll_ex1'),
    data: {
        label: '',
        onTopOrBottom: false
    },
    method: {
        onScroll: function() {
            if (this.onTopOrBottom) return;
            this.label = '<div style="color: #666666;">on scrolling...</div>';
        },
        onTop: function() {
            // to display message for a while
            this.onTopOrBottom = true;
            setTimeout(function() { this.onTopOrBottom = false; }.bind(this), 200);
            //
            this.label = '<div style="color: #ff0000; font-weight: bold;">scroll top hit!</div>';
        },
        onBottom: function() {
            // to display message for a while
            this.onTopOrBottom = true;
            setTimeout(function() { this.onTopOrBottom = false; }.bind(this), 200);
            //
            this.label = '<div style="color: #0000ff; font-weight: bold;">scroll bottom hit!</div>';
        }
    }
}).mount();
</script>

Condition

If / Neither



<div id="app_condition_if_ex1">
    <h3>Condition (if/neither)</h4>
    <div class="example">
        <label hc:if="visible">visible is true</label>
        <label hc:neither="visible">visible is false</label>
        <br />
        <button hc:click="setVisible(true)">set visible true</button>
        <button hc:click="setVisible(false)">set visible false</button>
    </div>
</div>

<script>
new Hacci({
    el: document.querySelector('#app_condition_if_ex1'),
    data: {
        visible: true
    },
    method: {
        setVisible: function(val) {
            this.visible = val;
        }
    }
}).mount();
</script>
